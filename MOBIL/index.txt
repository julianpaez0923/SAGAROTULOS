    <script>
        // DB Y CORE
        let db = JSON.parse(localStorage.getItem('SAGA_WOLF_DB')) || { productos: [], movs: [], currentType: 'ENTRADA', tempRem: [], rotulos: [] };
        const save = () => { localStorage.setItem('SAGA_WOLF_DB', JSON.stringify(db)); render(); updateKPIs(); };

        function nav(id) {
            document.querySelectorAll('.page-content').forEach(s => s.classList.remove('active'));
            document.querySelectorAll('.nav-item').forEach(b => b.classList.remove('active'));
            document.getElementById('sec-' + id).classList.add('active');
            document.getElementById('nav-' + id).classList.add('active');
            lucide.createIcons();
            if(id === 'remi' || id === 'labels') updateVisualPreviews();
        }

        function setMovType(t) {
            db.currentType = t;
            document.getElementById('btn-ent').className = t === 'ENTRADA' ? "flex-1 py-4 rounded-xl font-bold bg-green-600/20 border-2 border-green-600 text-green-500" : "flex-1 py-4 rounded-xl font-bold bg-white/5 text-slate-400";
            document.getElementById('btn-sal').className = t === 'SALIDA' ? "flex-1 py-4 rounded-xl font-bold bg-red-600/20 border-2 border-red-600 text-red-500" : "flex-1 py-4 rounded-xl font-bold bg-white/5 text-slate-400";
        }

        // OPERACIONES
        function ejecutarOperacion() {
            const proc = document.getElementById('m-proc').value.trim();
            const nom = document.getElementById('m-nom').value.trim();
            const cant = parseInt(document.getElementById('m-cant').value);
            const resp = document.getElementById('m-resp').value || 'SAGA_USER';

            if(!proc || !nom || isNaN(cant)) return alert("Complete los campos obligatorios.");

            let p = db.productos.find(x => x.nom.toLowerCase() === nom.toLowerCase());
            if(!p) {
                if(db.currentType === 'SALIDA') return alert("No existe stock previo.");
                p = { id: Date.now(), nom: nom, stk: 0 };
                db.productos.push(p);
            }
            if(db.currentType === 'SALIDA' && p.stk < cant) return alert("Stock insuficiente.");
            
            p.stk += (db.currentType === 'ENTRADA' ? cant : -cant);
            const m = { fecha: new Date().toLocaleString(), raw: new Date().toISOString().split('T')[0], tipo: db.currentType, proceso: proc, producto: p.nom, cantidad: cant, saldo: p.stk, resp: resp };
            db.movs.push(m);
            save();
            document.getElementById('m-nom').value = ""; document.getElementById('m-cant').value = "";
            alert("Operación Registrada.");
        }

        // LÓGICA DE REMISIONES (CORREGIDA)
        function addRemItem() {
            const search = document.getElementById('r-search').value;
            const cant = document.getElementById('r-cant').value;
            if(search && cant) {
                db.tempRem.push({ n: search, c: cant });
                document.getElementById('r-search').value = "";
                document.getElementById('r-cant').value = "";
                updateRemPreview();
            }
        }

        function updateRemPreview() {
            const cli = document.getElementById('r-cli').value || 'CLIENTE FINAL';
            const doc = document.getElementById('r-doc').value || 'NIT / CC';
            
            // Inyectar datos dinámicos
            document.getElementById('pv-r-cli').innerText = cli.toUpperCase();
            document.getElementById('pv-r-doc').innerText = "IDENTIFICACIÓN: " + doc;
            document.getElementById('pv-rem-num').innerText = "#RM-" + Date.now().toString().slice(-6);
            document.getElementById('pv-rem-fecha').innerText = new Date().toLocaleDateString();

            // Listado de productos en la hoja
            document.getElementById('pv-remi-items').innerHTML = db.tempRem.map(i => `
                <tr class="border-b border-slate-100">
                    <td class="p-3 font-bold text-slate-900">${i.c}</td>
                    <td class="p-3 uppercase text-slate-700">${i.n}</td>
                </tr>`).join('');

            // Listado lateral en el formulario
            document.getElementById('remi-list').innerHTML = db.tempRem.map((i, idx) => `
                <div class="flex justify-between items-center bg-white/5 p-2 rounded-lg border border-white/10">
                    <span class="text-xs">${i.c} x ${i.n}</span>
                    <button onclick="db.tempRem.splice(${idx},1);updateRemPreview()" class="text-red-500 font-bold px-2">×</button>
                </div>`).join('');
        }

        function resetRemision() { db.tempRem = []; document.getElementById('r-cli').value = ""; document.getElementById('r-doc').value = ""; signaturePad.clear(); updateRemPreview(); }
        function generarRemisionPDF() { html2pdf().from(document.getElementById('remi-sheet')).save('Remision_SagaWolf.pdf'); }

        // LÓGICA DE RÓTULOS (CORREGIDA)
        function addLabelToQueue() {
            const l = { 
                dest: document.getElementById('l-dest').value, 
                cont: document.getElementById('l-cont').value, 
                cant: document.getElementById('l-cant').value, 
                caja: document.getElementById('l-caja').value 
            };
            if(!l.dest || !l.cont) return alert("Destinatario y contenido son obligatorios.");
            db.rotulos.push(l);
            document.getElementById('l-cont').value = ""; 
            document.getElementById('l-cant').value = ""; 
            document.getElementById('l-caja').value = "";
            save(); updateLabelPreview();
        }

        function updateLabelPreview() {
            const grid = document.getElementById('pv-labels-grid');
            if(db.rotulos.length === 0) {
                grid.innerHTML = '<p class="text-slate-400 text-center mt-10 italic">No hay etiquetas agregadas</p>';
                return;
            }
            grid.innerHTML = db.rotulos.map((l, idx) => `
                <div class="enterprise-label">
                    <div class="flex justify-between items-start mb-4">
                        <div class="flex items-center gap-2">
                            <i data-lucide="package" class="w-4 text-blue-600"></i>
                            <p class="text-[10px] font-black uppercase tracking-tighter text-slate-500">Documento de Transporte</p>
                        </div>
                        <button onclick="db.rotulos.splice(${idx},1);save();updateLabelPreview()" class="text-red-500 no-print font-bold">BORRAR</button>
                    </div>
                    <div class="mb-4">
                        <h4 class="text-[9px] font-bold text-slate-400 uppercase">Destinatario</h4>
                        <p class="text-lg font-black text-slate-900 uppercase leading-none">${l.dest}</p>
                    </div>
                    <div class="grid grid-cols-2 gap-4 border-t border-slate-100 pt-3 mt-2">
                        <div>
                            <p class="text-[9px] font-bold text-slate-400 uppercase">Contenido</p>
                            <p class="text-[11px] text-slate-800 font-bold uppercase">${l.cont}</p>
                        </div>
                        <div class="text-right">
                            <p class="text-[9px] font-bold text-slate-400 uppercase">Referencia</p>
                            <p class="text-[11px] text-slate-800 font-bold italic">CAJA ${l.caja} - ${l.cant}</p>
                        </div>
                    </div>
                </div>`).join('');
            lucide.createIcons();
        }

        function clearLabels() { db.rotulos = []; save(); updateLabelPreview(); }
        function exportarLabelsPDF() { html2pdf().from(document.getElementById('labels-sheet')).save('Rotulos_Logistica.pdf'); }

        // RENDER GENERAL
        function render() {
            const kG = document.getElementById('f-k-gen').value.toLowerCase();
            document.getElementById('lista-kardex').innerHTML = db.movs.slice().reverse().filter(m => 
                m.producto.toLowerCase().includes(kG) || m.proceso.toLowerCase().includes(kG)
            ).map(m => `
                <tr class="border-b border-white/5">
                    <td class="p-4 text-slate-500">${m.fecha}</td>
                    <td class="font-bold ${m.tipo==='ENTRADA'?'text-green-500':'text-red-500'}">${m.tipo}</td>
                    <td class="text-blue-400 italic">${m.proceso}</td>
                    <td>${m.producto}</td>
                    <td>${m.cantidad}</td>
                    <td class="font-bold text-blue-200">${m.saldo}</td>
                    <td>${m.resp}</td>
                </tr>`).join('');
            
            document.getElementById('dash-log').innerHTML = db.movs.slice().reverse().slice(0, 5).map(m => `
                <div class="flex justify-between p-3 bg-white/5 rounded-xl border-l-4 ${m.tipo==='ENTRADA'?'border-green-500':'border-red-500'}">
                    <div><p class="text-xs font-bold text-blue-400 uppercase">${m.proceso}</p><p class="text-[10px] text-slate-500 italic">${m.producto}</p></div>
                    <span class="font-black text-xs self-center">${m.tipo === 'ENTRADA' ? '+' : '-'}${m.cantidad}</span>
                </div>`).join('');
            
            document.getElementById('datalist-prod').innerHTML = db.productos.map(p => `<option value="${p.nom}">`).join('');
        }

        function updateKPIs() {
            document.getElementById('kpi-items').innerText = db.productos.length;
            document.getElementById('kpi-movs').innerText = db.movs.length;
            document.getElementById('kpi-alerts').innerText = db.productos.filter(p => p.stk < 10).length;
        }

        function updateVisualPreviews() { updateRemPreview(); updateLabelPreview(); }

        let signaturePad;
        window.onload = () => { 
            signaturePad = new SignaturePad(document.getElementById('signature-pad'));
            render(); updateKPIs(); nav('dash'); 
        };
    </script>
